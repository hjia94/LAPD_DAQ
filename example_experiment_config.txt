[bmotion]
# Consumed only by Data_Run_bmotion.py;
# Data_Run_bmotion.py uses a separate TOML file for actual motion plane and movement
# Selects which motion groups from bmotion_config.toml
# If this section is omitted entirely: all motion groups run forward.
#
# motion_groups accepts:
#   - "all"                       -> every group in the TOML
#   - comma- or space-separated keys, e.g. "0, 2, 5"  or  "P22 P29"
#
# direction accepts:
#   - bare word "forward" / "backward" applied to ALL selected groups
#   - per-key mapping: "0=forward, 2=backward, 5=forward"
#   - keys not listed default to "forward"
#
# execution_order selects how multiple motion groups are run:
#   - "interleaved" (default): every motion group advances together; at each
#     motion-list index, every group moves to that index, then nshots are taken.
#   - "sequential": one motion group runs its full scan to completion before
#     the next motion group starts. Idle groups are not recorded during another
#     group's phase.
motion_groups = all
direction = forward
execution_order = interleaved
